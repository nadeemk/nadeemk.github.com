Binary Search

Iterative Solution 

/// Binary Search - Iterative Solution.
///
/// Copyright 2011, Nadeem Khan
///


int binary_search(int x, int[] a) 
{
	if(a == null || a.Length == 0) return -1;
	int l = 0, u = a.Length -1, mid;
	
	while(l <= u) 
	{
		mid = l + (u - l ) / 2;
		
		if( a[mid] == x ) 
		{
			return mid;
		}
		if( a[mid] <  x ) 
		{
			l = mid + 1;
		}
		else 
		{
			u = mid - 1;
		}
	}
	
	return -1;
}
