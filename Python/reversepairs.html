############### Core Algorithm #############
def reversePairs(t):
    if t == None or t.next == None: return t

    startOfNextPair = reversePairs(t.next.next)

    t.next.next = t
    startOfThisPair = t.next
    t.next = startOfNextPair

    return startOfThisPair
